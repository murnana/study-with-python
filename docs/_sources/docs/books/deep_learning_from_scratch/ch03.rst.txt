パーセプトロンの復習
=====================

.. graphviz::

    digraph perceptron {
        graph [
            charset = "UTF-8";
            rankdir = LR;
        ];

        x1 [label = <x<sub>1</sub>>];
        x2 [label = <x<sub>2</sub>>];
        y;

        x1 -> y [
            label = <w<sub>1</sub>>
        ];
        x2 -> y [
            label = <w<sub>2</sub>>
        ];
    }

.. math::

    \begin{split}
        y = \left\{ \begin{array}{ll}
            0 & (b + w_{1}x_{1} + w_{2}x_{2} \leqq 0) \\
            1 & (b + w_{1}x_{1} + w_{2}x_{2} > 0)
        \end{array} \right.
    \end{split}
